Segurança — Schema do Banco de Dados
Execute este SQL no SQL Editor do Supabase para criar as tabelas com Row Level Security (RLS) ativado.
-- ═══════════════════════════════════════ -- ML ENVIOS — Supabase Schema v1.0 -- Execute no SQL Editor do Supabase -- ═══════════════════════════════════════ -- 1. CÓDIGO DO DIA CREATE TABLE IF NOT EXISTS ml_daily_codes ( id uuid PRIMARY KEY DEFAULT gen_random_uuid(), seller_id TEXT NOT NULL, code TEXT NOT NULL, notes TEXT, code_date DATE NOT NULL DEFAULT CURRENT_DATE, updated_by TEXT, updated_at TIMESTAMPTZ DEFAULT now(), created_at TIMESTAMPTZ DEFAULT now(), UNIQUE (seller_id, code_date) ); CREATE INDEX IF NOT EXISTS idx_codes_seller_date ON ml_daily_codes (seller_id, code_date); ALTER TABLE ml_daily_codes ENABLE ROW LEVEL SECURITY; CREATE POLICY "anon_select_codes" ON ml_daily_codes FOR SELECT USING (true); CREATE POLICY "anon_insert_codes" ON ml_daily_codes FOR INSERT WITH CHECK (true); CREATE POLICY "anon_update_codes" ON ml_daily_codes FOR UPDATE USING (true) WITH CHECK (true); -- 2. CONFIRMAÇÕES DE ENVIO CREATE TABLE IF NOT EXISTS ml_confirmacoes ( id uuid PRIMARY KEY DEFAULT gen_random_uuid(), seller_id TEXT NOT NULL, order_id TEXT NOT NULL, confirmed BOOLEAN DEFAULT false, reason TEXT, notes TEXT, photo_url TEXT, shipping_date DATE NOT NULL DEFAULT CURRENT_DATE, updated_at TIMESTAMPTZ DEFAULT now(), created_at TIMESTAMPTZ DEFAULT now(), UNIQUE (seller_id, order_id, shipping_date) ); CREATE INDEX IF NOT EXISTS idx_confirms_seller_date ON ml_confirmacoes (seller_id, shipping_date); CREATE INDEX IF NOT EXISTS idx_confirms_order ON ml_confirmacoes (order_id); ALTER TABLE ml_confirmacoes ENABLE ROW LEVEL SECURITY; CREATE POLICY "anon_select_confirms" ON ml_confirmacoes FOR SELECT USING (true); CREATE POLICY "anon_insert_confirms" ON ml_confirmacoes FOR INSERT WITH CHECK (true); CREATE POLICY "anon_update_confirms" ON ml_confirmacoes FOR UPDATE USING (true) WITH CHECK (true); -- 3. STORAGE BUCKET INSERT INTO storage.buckets (id, name, public) VALUES ('fotos', 'fotos', true) ON CONFLICT (id) DO NOTHING; CREATE POLICY "public_read_fotos" ON storage.objects FOR SELECT USING (bucket_id = 'fotos'); CREATE POLICY "anon_upload_fotos" ON storage.objects FOR INSERT WITH CHECK (bucket_id = 'fotos'); -- 4. REALTIME (habilitar nas tabelas) -- Vá em Database → Replication e ative: -- ml_daily_codes, ml_confirmacoes
